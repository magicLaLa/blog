# ES6的一些方法和技巧

# 声明变量

# 用let不用var

ES6之前我们用var声明一个变量，但是它有很多弊病：

  • 因为没有块级作用域，很容易声明全局变量
  • 变量提升
  • 可以重复声明 还记得这道面试题吗？
var a = [];
for (var i = 0; i < 10; i++) {
  a[i] = function () {
    console.log(i);
  };
}
a[6](); // 10
a[7](); // 10
a[8](); // 10
a[9](); // 10

所以要使用 let，可以避免 使用 var 来声明所带来的一些问题。

# const 和 let

const 和 let 的区别就是 使用 const 来声明的变量，称为 常量

  • 更好的代码语义化，哪个可以修改，哪个不可以修改等等
  • 函数声明应该使用 const 来声明

# 字符串

不要使用 "双引号"，一律使用 '单引号' 或者 `模板字符串`

模板字符串可以当作普通字符串使用，也可以用来定义多行字符串，或者在字符串中嵌入变量

// low
const a = "123";
let b = "a" + b + "c";

// good
const str = '456'
let arrStr = `123${ 变量/JavaScript 表达式 }456`

之前判断字符串是否在另一个字符串中存在，用的是 indexOf 来判断，在 ES6 中提供了三个新的方法:

  • includes()：返回布尔值，表示是否找到了参数字符串。
  • startsWith()：返回布尔值，表示参数字符串是否在原字符串的头部。
  • endsWith()：返回布尔值，表示参数字符串是否在原字符串的尾部。
let s = 'Hello world!';

s.startsWith('Hello') // true
s.endsWith('!') // true
s.includes('o') // true

这三个方法都支持第二个参数，表示开始搜索的位置。

let s = 'Hello world!';

s.startsWith('world', 6) // true
s.endsWith('Hello', 5) // true
s.includes('Hello', 6) // fals
  • repeat()方法返回一个新字符串，表示将原字符串重复n次。
'-'.repeat(10) // '----------'
// 可以在 console.log() 里面来使用，区分上下

# 结构赋值

# 变量赋值

在用到数组成员对变量赋值时，尽量使用解构赋值。

const arr = [1, 2, 3, 4];

// low
const first = arr[0];
const second = arr[1];

// good
const [first, second] = arr;

# 函数参数为对象时

函数的参数如果是对象的成员，优先使用解构赋值。

// low
function getFullName(user) {
  const firstName = user.firstName;
  const lastName = user.lastName;
}

// good
function getFullName({ firstName, lastName }) {
  ...
}

如果函数返回多个值，优先使用对象的解构赋值，而不是数组的解构赋值。这样便于以后添加返回值，以及更改返回值的顺序。

// low
function processInput(input) {
  return [left, right, top, bottom];
}

// good
function processInput(input) {
  return { left, right, top, bottom };
}

const { left, right } = processInput(input);

# 对象Object

# 对象属性名是动态时，可以使用 属性表达式

// low
const obj = {
  id: 5,
  name: 'San Francisco',
};
obj[getKey('enabled')] = true;

// good
const obj = {
  id: 5,
  name: 'San Francisco',
  [getKey('enabled')]: true,
};

# 再简洁一点

在定义对象时，能简洁表达尽量简洁表达：

let name = 'John';

// low
let obj = {
  name: name,
  age: 18,
  sleep: function () {
    console.log(`sleep`)
  }
}

// good
let obj = {
  name,
  age: 18,
  sleep: function () {
    console.log(`sleep`)
  }
}

# 简单的键值对优先Map

如果只是简单的key: value结构，建议优先使用Map，因为Map提供方便的遍历机制。

let map = new Map(obj);
// 遍历key值
for (let key of map.keys()) {
  console.log(key);
}
// 遍历value值
for (let value of map.values()) {
  console.log(value);
}
// 遍历key和value值
for (let item of map.entries()) {
  console.log(item[0], item[1]);
}

# 数组

# ... 扩展运算符

使用扩展运算符（...）拷贝数组：

// 还在用for i 你就太low了
const len = items.length;
const itemsCopy = [];
let i;

for (i = 0; i < len; i++) {
  itemsCopy[i] = items[i];
}

// cool !
const itemsCopy = [...items];

// 可以结合 Set构造函数 来对数组进行去重
let a = [...new Set([1,1,2,5,8,3,2,1])]
// [1, 2, 5, 8, 3]

# 类数组

可以用 Array.from 将类似数组的对象转为数组：

const arr = Array.from(arguments);

# 函数

# 参数设置默认值

// low
function add (x, y) {
  x = x || 0;
  y = y || 0;
}

// good
function add (x, y = '1') {
  return x + y
}

# rest参数 (...变量名)

ES6 引入 rest 参数（形式为...变量名），用于获取函数的多余参数，这样就不需要使用arguments对象了。rest 参数搭配的变量是一个数组，该变量将多余的参数放入数组中。

// low
function sortNum() {
  let args = Array.prototype.slice.call(arguments);
  return args.sort()
}

// good
function sortNum(...args) {
  return args.sort()
}

# 箭头函数 =>

  • 立即执行函数
(() => {
  ...
  console.log('***')
})()
  • 可以使代码变得简洁优雅
// low
[1, 2, 3].map(function (x) {
  return x * x;
});

// cool !
[1, 2, 3].map(x => x * x);
  • 箭头函数只有一行语句，且不需要返回值，可以采用下面的写法，就不用写大括号了
let fn = () => void doesNotReturn();
  • 箭头函数可以与变量解构结合使用
const full = ({ first, last }) => first + ' ' + last;

// 等同于
function full(person) {
  return person.first + ' ' + person.last;
}

# async、await

async 函数返回的是一个 Promise 对象，可以在后面.then((res)) => {})来添加回调函数，内部遇到await会等待返回结果，await后面是一个 Promise 对象，如果不是，会被转成一个立即resolve的 Promise 对象。

async 函数内部报错时，可以被.catch()捕获。

async function add(reqBody) {
  let data = await fetch({
    url: '...',
    data: {
      reqBody: reqBody
    }
  })
  return data
}

add({a, b}).then(res => {
  console.log(res)
})

# 简洁直观的class语法

// low
function Query(arr1, arr2) {
  this._arr = [...arr1, ...arr2] // 合并数组
}
Query.prototype.push = function (item) {
  this._arr = this._arr.splice(this._arr.length, 0, item)
}

// good
class Query {
  constructor (arr1, arr2) {
    this._arr = [...arr1, ...arr2]
  }

  push (item) {
    this._arr = this._arr.splice(this._arr.length, 0, item)
  }
}

# ES6模块化

# 引入模块 和 输出模块

// 第一种
export const func1 = function () {...}
import { func1 } from 'moduleA'

// 第二种
export default function func2 () {...}
import func1 from 'moduleA'

// 第三种
export const func1 = function () {...}
export const func2 = function () {...}
export const func3 = function () {...}
import { func1, func2, func3 } from 'moduleA'
// 或者
import * as types from 'moduleA'
// types.BUILD_CITYTREE
  • 输出单个值，使用export default
  • 输出多个值，使用export
  • export default与普通的export不要同时使用

localStorage 统一管理设置-增加时间

Written by Magic with ♥ on 3 December 2018 in js 技术相关随笔 1 min
  • 设置本地存储
function setStorage(key: string, value: any) {
  try {
    localStorage.setItem(key,  JSON.stringify({
      data: value,
      time: Date.now(),
    }));
  } catch {
    console.log('localStorage.setItem error');
    console.log(key, value);
  }
}
  • 获取本地存储
// 注意：stringify 会过滤掉undefined的值
function getStorage(key: string) {
  const result = localStorage.getItem(key);
  if (result) {
    try {
      const objRes = JSON.parse(result);
      return objRes;
    } catch (e) {
      console.log('JSON.parse error:', result);
      return result;
    }
  } else {
    return result;
  }
}
  • 删除本地存储
function removeStorage(key: string) {
  localStorage.removeItem(key);
}

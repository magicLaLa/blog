移除链表元素

  • 删除链表中等于给定值 val 的所有节点。
示例:
输入: 1->2->6->3->4->5->6, val = 6
输出: 1->2->3->4->5

分析

/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var removeElements = function(head, val) {
  let tem = head;
  let pre = null;
  while (tem) {
    if (tem.val === val) {
      if (pre) {
        pre.next = tem.next;
      } else {
        head = tem.next;
      }
    } else {
      pre = tem;
    }
    tem = tem.next;
  }
  return head;
};

var removeElements = function(head, val) {
  if (!head) return null;
  if (head.val === val) return removeElements(head.next, val);
  head.next = removeElements(head.next, val);
  return head;
};

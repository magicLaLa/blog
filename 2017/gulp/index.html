gulpfile.js

var gulp = require('gulp')
var plugins = require('gulp-load-plugins')() // 加载 gulp 插件
var browserSync = require('browser-sync').create() // 自动刷新
var reload = browserSync.reload
var merge = require('merge-stream')
var htmlmin = require('gulp-html-minifier') // 压缩 html
var sass = require('gulp-sass') // 编译 sass
var spritesmith = require('gulp.spritesmith') // 雪碧图
var buffer = require('vinyl-buffer')
var imagemin = require('gulp-imagemin') // 压缩图片
var pngquant = require('imagemin-pngquant') // 压缩 png
var mozjpeg = require('imagemin-mozjpeg') // 压缩 jpg
var runSequence = require('run-sequence') // 序列
var dateFormat = require('dateformat') // 日期格式化
var now = new Date() // 当前时间

var glob = require('glob')

var remUnit = 46.875 // 根元素字体大小 = 设计宽度 / 320 * 20

// 获取页面文件名
function getPagesName () {
  var pages = []
  glob.sync('src/**/pages/').forEach(function (filePath) {
    var page = filePath.substring('src/'.length, filePath.indexOf('/pages/'))
    pages.push(page)
  })
  pages.push('public')
  return pages
}

var pagesName = getPagesName()

// 复制 html
gulp.task('html', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/pages'
    return gulp.src('src/' + pageName + '/pages/**/*.html')
      .pipe(plugins.changed(DEST))
      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 压缩 html
gulp.task('compress-html', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/pages'

    return gulp.src('src/' + pageName + '/pages/**/*.html')
      // 压缩 html
      .pipe(htmlmin({
        collapseWhitespace: true,
        minifyJS: true,
        minifyCSS: true,
        removeComments: true
      }))

      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 处理 css
gulp.task('css', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/css'

    return gulp.src('src/' + pageName + '/sass/**/*.scss')
      .pipe(plugins.sourcemaps.init())
      // 转换 sass
      .pipe(sass().on('error', sass.logError))
      // px 转 rem
      .pipe(plugins.pxtorem({
        rootValue: remUnit,
        propList: ['*']
      }, {}))
      // 自动前缀
      .pipe(plugins.autoprefixer({
        browsers: ['Android >= 2.3', 'iOS >= 4.3']
      }))
      .pipe(plugins.sourcemaps.write())
      .pipe(gulp.dest(DEST))
      // 注入 browserSync
      .pipe(browserSync.stream())
  })
  return merge(tasks)
})

// 压缩 css
gulp.task('compress-css', function () {
  var tasks = pagesName.map(function (pageName) {
    return gulp.src('dist/xxxx/' + pageName + '/css/**/*.css', {base: '.'})
      // 压缩 css
      .pipe(plugins.cssnano({
        reduceIdents: false, // 不压缩动画名字
        autoprefixer: false, // 不需要自动前缀
        zindex: false // 不压缩z-index
      }))

      .pipe(gulp.dest('.'))
  })
  return merge(tasks)
})

// 复制 js
gulp.task('js', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/js'
    return gulp.src('src/' + pageName + '/js/**/*.js')
      .pipe(plugins.changed(DEST))
      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 压缩 js
gulp.task('compress-js', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/js'
    return gulp.src('src/' + pageName + '/js/**/*.js')
      // 压缩 js
      .pipe(plugins.uglify())

      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 复制图片
gulp.task('img', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/images'

    return gulp.src('src/' + pageName + '/img/**/*.{jpg,png,gif,svg}')
      .pipe(plugins.changed(DEST))
      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 压缩图片
gulp.task('compress-img', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/images'

    return gulp.src('src/' + pageName + '/img/**/*.{jpg,png,gif,svg}')
      // 压缩图片
      .pipe(imagemin([
        pngquant({quality: '60'}),
        mozjpeg({quality: 60})
      ]))
      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 雪碧图
gulp.task('sprite', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/images'

    var spriteData = gulp.src('src/' + pageName + '/sprite/*.{png,gif}')
      .pipe(plugins.changed(DEST))
      .pipe(spritesmith({
        imgName: 'sprite.png',
        cssName: '_sprite.scss',
        cssTemplate: 'scss.template.handlebars',
        // 图片间距
        padding: 6
      }))

    var imgStream = spriteData.img

      .pipe(buffer())
      .pipe(imagemin([
        pngquant({quality: '60'})
      ]))
      .pipe(gulp.dest(DEST))

    var cssStream = spriteData.css
      .pipe(gulp.dest('src/' + pageName + '/sass'))

    return merge(imgStream, cssStream)
  })
  return merge(tasks)
})

// 删除 dist
gulp.task('clean', function () {
  return gulp.src(['dist', '【打包输出】'], {read: false})
    .pipe(plugins.clean())
})

// 合并
gulp.task('all', function () {
  runSequence('sprite', ['html', 'css', 'img', 'js'], 'manifest')
})

// 生成压缩包(正式)
gulp.task('zip', function () {
  return gulp.src('dist/xxxx/**/*', {base: 'dist'})
    .pipe(plugins.zip(dateFormat(now, 'yyyymmddHHMM') + 'xxxx.zip'))
    .pipe(gulp.dest('【打包输出】'))
})

// 生成压缩包(测试)
gulp.task('zip:test', function () {
  return gulp.src('dist/xxxx/**/*', {base: 'dist'})
    .pipe(plugins.zip(dateFormat(now, 'yyyymmddHHMM') + 'xxxxtest.zip'))
    .pipe(gulp.dest('【打包输出】'))
})

// 上传压缩包
gulp.task('sftp', function () {
  return gulp.src('【打包输出】/*xxxx*.zip', {base: '【打包输出】'})
    .pipe(plugins.sftp({
      host: 'xxxxxxx',
      user: 'xxxxx',
      pass: 'xxxxxxx',
      remotePath: 'xxxx/xxxx/xxx/xx'
    }))
})

gulp.task('manifest', function () {
  var tasks = pagesName.map(function (pageName) {
    const DEST = 'dist/xxxx/' + pageName + '/manifest'

    return gulp.src(['dist/xxxx/' + pageName + '/**/*', 'dist/xxxx/public/**/*', '!dist/xxxx/**/manifest/*'], {base: 'dist/xxxx/' + pageName + '/**'})
      .pipe(plugins.manifest({
        hash: true,
        network: ['*'],
        filename: 'app.manifest',
        exclude: 'app.manifest'
      }))
      .pipe(gulp.dest(DEST))
  })
  return merge(tasks)
})

// 输出 dist 文件
gulp.task('dist', function () {
  runSequence('clean', ['compress-html', 'compress-js', 'compress-img', 'css'], 'compress-css', 'sprite')
})

// 接口地址改成正式环境
gulp.task('replace', function () {
  return gulp.src('./dist/xxxx/public/js/publicClient.js', {base: './'})
    .pipe(plugins.replace('https://xxxxxx', 'https://xxxxxxx'))
    .pipe(gulp.dest('.'))
})

// 接口地址改成测试环境
gulp.task('replace:test', function () {
  return gulp.src('./dist/xxxx/public/js/publicClient.js', {base: './'})
    .pipe(plugins.replace('https://xxxxxxx', 'https://xxxxxxxxx'))
    .pipe(gulp.dest('.'))
})

// 打包（正式）
gulp.task('db', function () {
  runSequence('replace', 'zip')
})

// 打包（测试）
gulp.task('db:test', function () {
  runSequence('replace:test', 'zip:test')
})

gulp.task('dev', function () {
  browserSync.init({
    server: {
      baseDir: './dist'
    },
    open: 'external'
  })

  gulp.watch('dist/**/*.html').on('change', reload)
  gulp.watch('dist/**/*.js').on('change', reload)
  gulp.watch('dist/**/*.{png,gif,jpg}').on('change', reload)
  gulp.watch('dist/**/*.{html,js,css,png,jpg,gif}', ['manifest'])
  gulp.watch('src/**/pages/*.html', ['html'])
  gulp.watch('src/**/js/*.js', ['js'])
  gulp.watch('src/**/sprite/*.{png,gif}', ['sprite'])
  gulp.watch('src/**/sass/*.scss', ['css'])
})

gulp.task('revJs', function () {
  var tasks = pagesName.map(function (pageName) {
    return gulp.src('./dist/xxxx/' + pageName + '/js/*.js', {base: './dist'})
      .pipe(plugins.rev()) // 给文件添加hash编码
      .pipe(gulp.dest('./dist'))
      .pipe(plugins.rev.manifest()) // 生成rev-mainfest.json文件作为记录
      .pipe(gulp.dest('./dist/xxxx/' + pageName + '/rev/js'))
  })
  return merge(tasks)
})

gulp.task('revHtmlJs', function () {
  var tasks = pagesName.map(function (pageName) {
    return gulp.src(['./dist/xxxx/**/rev/js/*.json', './dist/xxxx/' + pageName + '/pages/*.html'])
      .pipe(plugins.revCollector())
      .pipe(gulp.dest('./dist/xxxx/' + pageName + '/pages/'))
  })
  return merge(tasks)
})

gulp.task('revCss', function () {
  var tasks = pagesName.map(function (pageName) {
    return gulp.src('./dist/xxxx/' + pageName + '/css/*.css', {base: './dist'})
      .pipe(plugins.rev()) // 给文件添加hash编码
      .pipe(gulp.dest('./dist'))
      .pipe(plugins.rev.manifest()) // 生成rev-mainfest.json文件作为记录
      .pipe(gulp.dest('./dist/xxxx/' + pageName + '/rev/css'))
  })
  return merge(tasks)
})

gulp.task('revHtmlCss', function () {
  var tasks = pagesName.map(function (pageName) {
    return gulp.src(['./dist/xxxx/**/rev/css/*.json', './dist/xxxx/' + pageName + '/pages/*.html'])
      .pipe(plugins.revCollector())
      .pipe(gulp.dest('./dist/xxxx/' + pageName + '/pages/'))
  })
  return merge(tasks)
})

gulp.task('rev', function () {
  runSequence(['revJs', 'revCss'], ['revHtmlJs', 'revHtmlCss'], 'manifest')
})


// 灰度打包
gulp.task('copy:gray', function () {
  return gulp.src('dist/**/*')
    .pipe(plugins.copy('dist/xxxx_tmp', {prefix: 2}))
})
gulp.task('replace:gray', function () {
  return gulp.src('dist/xxxx_tmp/**/*.html', {base: './'})
    .pipe(plugins.replace('/xxxx/', '/xxxx_tmp/'))
    .pipe(gulp.dest('.'))
})
gulp.task('zip:gray', function () {
  return gulp.src('dist/xxxx_tmp/**/*', {base: 'dist'})
    .pipe(plugins.zip(dateFormat(now, 'yyyymmddHHMM') + 'xxxx_tmp.zip'))
    .pipe(gulp.dest('【打包输出】'))
})
gulp.task('db:gray', function () {
  runSequence('copy:gray', 'replace:gray', 'zip:gray')
})

  • 注:有些gulp插件是通过plugins来调用的，需要自行安装gulp-xxx，然后再通过plugins.xxxx来调用

  • 使用方法

# 安装依赖
yarn

# 开发，注意需要先有自己的工程目录才能启动开发服务
yarn dev

# 打包（正式）
yarn db

# 打包（灰度）
yarn db:gray

# 打包（测试）
yarn db:test

Two Sum 两数之和(1)

  • 给定一个整数数组，返回两个数字的索引，使它们相加到特定目标。您可以假设每个输入只有一个解决方案，并且您可能不会两次使用相同的元素。
  • 例子： 
Given nums = [2, 7, 11, 15], target = 9,

Because nums[0] + nums[1] = 2 + 7 = 9,
return [0, 1].
  • 解:
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
  // 第一种
  let map = new Map()
  for (let i = 0; i < nums.length; i++) {
    if (map.has(target - nums[i])) {
      return [map.get(target - nums[i]), i];
    } else {
      map.set(nums[i], i);
    }
  }
  // 第二种
  const map = {};
  for (let i = 0; i < nums.length; i++) {
    const comp = target-nums[i];
    if (comp in map) {
      return [map[comp], i];
    }
    map[nums[i]] = i;
  }
};

反转整数 Reverse Integer(7)

  • 给定32位有符号整数，整数的反向数字

  • 例子:

  Input: 123
  Output: 321

  Input: -123
  Output: -321

  Input: 120
  Output: 21
  • 解:
/**
 * @param {number} x
 * @return {number}
 */
var reverse = function(x) {
  // 第一种
  let maxInteger = Math.pow(2, 31) - 1;
  let str = ''
  if (x === 0) return 0
  if (x < 0) {
      str = '-'
      x = Math.abs(x);
  }
  let list = x.toString().split('');
  list.reverse()
  if (list[0] === 0) {
      list.shift()
  }
  let num = Number(list.join('')) > maxInteger ? 0 : Number(list.join(''))
  return Number(str + num)

  // 第二种
  let y = Math.abs(x);
  if(y.toString(2).length > 31) return 0;
  let arr = y.toString().split("").reverse();
  if(Number(arr.join("")).toString(2).length > 31) return 0;
  let z = Number(arr.join(""));
  return x >= 0 ? z : -z;

  // 第三种
  let res = 0;
  let y = Math.abs(x);
  while(y > 0){
      res = res * 10 + y % 10;
      y = y / 10 | 0; // 按位或
  }
  if(res > Math.pow(2, 31) + 1 || x < Math.pow(-2, 31)) return 0;
  return x > 0 ? res : -res;
};

回文数 Palindrome Number(9)

  • 判断一个整数是否是回文数。回文数是指正序（从左向右）和倒序（从右向左）读都是一样的整数。

  • 例子:

示例 1:
  输入: 121
  输出: true

示例 2:
  输入: -121
  输出: false
  解释: 从左向右读, 为 -121 。 从右向左读, 为 121- 。因此它不是一个回文数。

示例 3:
  输入: 10
  输出: false
  解释: 从右向左读, 为 01 。因此它不是一个回文数。
  • 解:
/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function(x) {
  // 第一种
  let a = x.toString().split('');
  a.reverse();
  let n = a.join('');
  if (x == n) {
      return true
  } else {
      return false
  }
};
